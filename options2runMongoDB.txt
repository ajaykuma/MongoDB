#Running MongoDB as standalone instance(bound to Ip Address of machine)
when using deb package and running with defaults
mongod -f /etc/mongod.conf
#data path = /var/lib/mongodb/
#log path = /var/log/mongodb/mongod.log
mongo --host hostname


#when running without replicaset with ipaddress
mng$sudo mkdir /data/db
mng$sudo chown -R mng:mng /data/db
mng$sudo mkdir /var/log/mongodb
mng$sudo chown -R mng:mng /var/log/mongodb
mng$mongod --dbpath /data/db --logpath /var/log/mongodb/mongod.log --bind_ip=192.168.xx.xx

--to check log
mng$tail -f /var/log/mongodb/mongod.log

--to connect
mng$mongo --host 192.168.xx.xx


#Running MongoDB as a member of replicaset
mongod --dbpath /srv/mongodb/db2 --replSet ds0 --bind_ip 10.0.3.1xx
