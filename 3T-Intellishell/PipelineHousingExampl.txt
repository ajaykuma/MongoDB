db.getCollection("Housing").aggregate(

	// Pipeline
	[
		// Stage 1
		{
			$match: {
			    "Property Type":"Senior"
			}
		},

		// Stage 2
		{
			$group: {
			    _id: "$Zip Code",
			    totalUnits: {$sum: "$Units"}
			}
		},

		// Stage 3
		{
			$sort: {
			    totalUnits: -1
			}
		},

	]

	// Created with Studio 3T, the IDE for MongoDB - https://studio3t.com/

);
